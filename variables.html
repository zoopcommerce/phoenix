Collection

Accessible from:
  • collection.twig
var collection = {
    appendHtml: string,
    children: array | [collection],
    depth: integer,
    display: boolean,
    images: array | [
        {
            raw: {
                alt: string,
                mime: string, // http://en.wikipedia.org/wiki/Internet_media_type#Type_image
                src: string,
                height: integer,
                width: integer
            },
            large: {
                alt: string,
                mime: string, // http://en.wikipedia.org/wiki/Internet_media_type#Type_image
                src: string,
                height: integer,
                width: integer
            },
            medium: {
                alt: string,
                mime: string, // http://en.wikipedia.org/wiki/Internet_media_type#Type_image
                src: string,
                height: integer,
                width: integer
            }
        }
    ],
    name: string,
    description: string,
    products: array | [product],
    pagination: {
        pages: array | [
            {
                active: boolean,
                disable: boolean,
                title: string,
                url: string
            }
        ]
    },
    prependHtml: string,
    url: {
        absolute: string, // The absolute url of the current page
        relative: string // The relative url of the current page
    }
}

Page

Accessible from:
  • all pages
var page = {
    css: array | [string], // contains an array of css files. This only applies if you are using Zoop default templates.
    dojo: array | [string], // contains an array of dojo javascript files. This only applies if you are using Zoop default templates.
    favicon: string | null, // favourite icon
    meta: {
        description: string // The page meta description,
        openGraph: string, // This contains open graph information https://developers.facebook.com/docs/opengraph/
    },
    title: string, // The page title which is made up from product, category and site name,
    javascript: array | [string], // contains an array of javascript files. This only applies if you are using Zoop default templates.
    url: {
        absolute: string, // The absolute url of the current page
        relative: string // The relative url of the current page
    }
}

Site

The site global variable contains information pertaining to the site.

Accessible from:
  • all pages
var site = {
    status: {
        active: boolean, // Is the store active or not
        message: string | null // Contains a message which can be displayed when a store is inactive
    }
}

    email: string,
    googleAnalyticsCode: string,

            url: string,

            url: string,
